-- Insert Permissions
DO $$
DECLARE
    permissions TEXT[][] := ARRAY[
        -- User permissions
        ARRAY['users:read', 'View users and their details'],
        ARRAY['users:create', 'Create new users'],
        ARRAY['users:update', 'Update existing users'],
        ARRAY['users:delete', 'Delete users from the system'],
        
        -- Role permissions
        ARRAY['roles:read', 'View roles and their details'],
        ARRAY['roles:create', 'Create new roles'],
        ARRAY['roles:update', 'Update existing roles'],
        ARRAY['roles:delete', 'Delete roles from the system'],
        
        -- Permission permissions
        ARRAY['permissions:read', 'View permissions and their details'],
        ARRAY['permissions:create', 'Create new permissions'],
        ARRAY['permissions:update', 'Update existing permissions'],
        ARRAY['permissions:delete', 'Delete permissions from the system'],
        ARRAY['permissions:assign', 'Assign permissions to roles'],
        
        -- Profile permissions
        ARRAY['profile:read', 'View own profile'],
        ARRAY['profile:update', 'Update own profile'],
        
        -- System permissions
        ARRAY['system:settings', 'Manage system settings'],
        ARRAY['system:logs', 'View system logs'],
        ARRAY['system:backup', 'Backup and restore system data']
    ];
    perm TEXT[];
BEGIN
    FOREACH perm SLICE 1 IN ARRAY permissions
    LOOP
        IF NOT EXISTS (SELECT 1 FROM "Permission" WHERE name = perm[1]) THEN
            INSERT INTO "Permission" (id, name, description, "createdAt", "updatedAt")
            VALUES (gen_random_uuid()::text, perm[1], perm[2], CURRENT_TIMESTAMP, CURRENT_TIMESTAMP);
        END IF;
    END LOOP;
END $$;

-- Assign permissions to roles
DO $$
DECLARE
    admin_role_id TEXT;
    manager_role_id TEXT;
    user_role_id TEXT;
    guest_role_id TEXT;
    perm_id TEXT;
    admin_perms TEXT[] := ARRAY[
        'users:read', 'users:create', 'users:update', 'users:delete',
        'roles:read', 'roles:create', 'roles:update', 'roles:delete',
        'permissions:read', 'permissions:create', 'permissions:update', 'permissions:delete', 'permissions:assign',
        'profile:read', 'profile:update',
        'system:settings', 'system:logs', 'system:backup'
    ];
    manager_perms TEXT[] := ARRAY[
        'users:read', 'users:create', 'users:update',
        'roles:read',
        'permissions:read',
        'profile:read', 'profile:update'
    ];
    user_perms TEXT[] := ARRAY[
        'profile:read', 'profile:update'
    ];
    guest_perms TEXT[] := ARRAY[
        'profile:read'
    ];
    perm_name TEXT;
BEGIN
    -- Get role IDs
    SELECT id INTO admin_role_id FROM "Role" WHERE name = 'Admin' LIMIT 1;
    SELECT id INTO manager_role_id FROM "Role" WHERE name = 'Manager' LIMIT 1;
    SELECT id INTO user_role_id FROM "Role" WHERE name = 'User' LIMIT 1;
    SELECT id INTO guest_role_id FROM "Role" WHERE name = 'Guest' LIMIT 1;
    
    -- Assign Admin permissions
    FOREACH perm_name IN ARRAY admin_perms
    LOOP
        SELECT id INTO perm_id FROM "Permission" WHERE name = perm_name LIMIT 1;
        IF perm_id IS NOT NULL AND NOT EXISTS (
            SELECT 1 FROM "RolePermission" 
            WHERE "roleId" = admin_role_id AND "permissionId" = perm_id
        ) THEN
            INSERT INTO "RolePermission" (id, "roleId", "permissionId", "createdAt", "updatedAt")
            VALUES (gen_random_uuid()::text, admin_role_id, perm_id, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP);
        END IF;
    END LOOP;
    
    -- Assign Manager permissions
    FOREACH perm_name IN ARRAY manager_perms
    LOOP
        SELECT id INTO perm_id FROM "Permission" WHERE name = perm_name LIMIT 1;
        IF perm_id IS NOT NULL AND NOT EXISTS (
            SELECT 1 FROM "RolePermission" 
            WHERE "roleId" = manager_role_id AND "permissionId" = perm_id
        ) THEN
            INSERT INTO "RolePermission" (id, "roleId", "permissionId", "createdAt", "updatedAt")
            VALUES (gen_random_uuid()::text, manager_role_id, perm_id, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP);
        END IF;
    END LOOP;
    
    -- Assign User permissions
    FOREACH perm_name IN ARRAY user_perms
    LOOP
        SELECT id INTO perm_id FROM "Permission" WHERE name = perm_name LIMIT 1;
        IF perm_id IS NOT NULL AND NOT EXISTS (
            SELECT 1 FROM "RolePermission" 
            WHERE "roleId" = user_role_id AND "permissionId" = perm_id
        ) THEN
            INSERT INTO "RolePermission" (id, "roleId", "permissionId", "createdAt", "updatedAt")
            VALUES (gen_random_uuid()::text, user_role_id, perm_id, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP);
        END IF;
    END LOOP;
    
    -- Assign Guest permissions
    FOREACH perm_name IN ARRAY guest_perms
    LOOP
        SELECT id INTO perm_id FROM "Permission" WHERE name = perm_name LIMIT 1;
        IF perm_id IS NOT NULL AND NOT EXISTS (
            SELECT 1 FROM "RolePermission" 
            WHERE "roleId" = guest_role_id AND "permissionId" = perm_id
        ) THEN
            INSERT INTO "RolePermission" (id, "roleId", "permissionId", "createdAt", "updatedAt")
            VALUES (gen_random_uuid()::text, guest_role_id, perm_id, CURRENT_TIMESTAMP, CURRENT_TIMESTAMP);
        END IF;
    END LOOP;
END $$;
